## Loading

<Playground>
  {() => {
    const [loading, setLoading] = React.useState(false);
    const onClick = () => {
      setLoading(true);
      setTimeout(() => setLoading(false), 1000);
    };
    return (
      <Inline space="m" alignY="center">
        <Button onClick={onClick} loading={loading}>
          Start loading
        </Button>
        <Button onClick={onClick} loading={loading} variant="flat">

        </Button>

# Checkbox

## Basic usage

```tsx
import { Checkbox } from '@fxtrot/edge';
```

<Playground>
  {() => {
    const [checked, setChecked] = React.useState(false);
    return (
      <Checkbox
        label="Subscribe to news"
        checked={checked}
        onChange={(e) => setChecked(e.target.checked)}
        message="This is the message"
      />
    );
  }}
</Playground>

# Inline

Wraps its children in a row.

## Basic usage

```tsx
import { Inline } from '@foxtrot/edge';
```

<Playground>
  <Box style={{ width: '400px' }}>
    <Inline>
      {Array(6)
        .fill(1)
        .map((_v, i) => (
          <Placeholder key={i} height={50} width={50}>
            {i + 1}
          </Placeholder>
        ))}
    </Inline>
  </Box>
</Playground>

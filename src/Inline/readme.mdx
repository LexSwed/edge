import { Playground } from 'docz';
import { Box, Inline, Placeholder } from '@fxtrot/edge';

# Inline

Wraps its children in a row.

## Basic usage

```tsx
import { Inline } from '@fxtrot/edge';
```

<Playground>
  <Box p="m" width={350}>
    <Inline space="m">
      {Array(7)
        .fill(1)
        .map((_v, i) => (
          <Placeholder key={i} height={50} width={50}>
            {i + 1}
          </Placeholder>
        ))}
    </Inline>
  </Box>
</Playground>

## Nowrap

<Playground>

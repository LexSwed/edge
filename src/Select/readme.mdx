# Select

## Basic usage

```tsx
import { Select, Option } from '@foxtrot/edge';
```

<Playground>
  {() => {
    const people = ['Taran Burt', 'Jaimee Rios', 'Haniya Padilla'];
    const [value, setValue] = React.useState(people[0]);
    return (
      <Box width={240}>
        <Select
          value={value}
          onSelect={(newValue) => setValue(newValue)}
          placeholder="Jogn Doe"
          label="Select assignee"
          message="This person will be assigned to the task"
        >
          {people.map((person) => (
            <Option key={person} value={person}>
              {person}
            </Option>
          ))}
        </Select>
      </Box>
    );
  }}
</Playground>
